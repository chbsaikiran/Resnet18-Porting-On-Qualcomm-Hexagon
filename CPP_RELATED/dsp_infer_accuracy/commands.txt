for QNN:

QAIRT_ROOT:/opt/qcom/aistack/qairt/2.31.0.250130 in ~/.bashrc

conda activate mlpy310env
export PYTHONPATH="${QAIRT_ROOT}/lib/python"

python onnx_simplified.py

python make_nhwc_input_onnx.py --in model_simplified.onnx --out model_nhwc_input.onnx

qnn-onnx-converter \
  --input_network model_nhwc_input.onnx \
  --output_path model \
  -d input_nhwc 1,32,32,3 \
  --input_layout input_nhwc NHWC \
  --out_node output

mv model model.cpp


qnn-model-lib-generator \
  -c model.cpp \
  -b model.bin \
  -l resnet18_cifar10 \
  -o ./lib \
  -t aarch64-android

./deploy_qnn.sh

./run_qnn_infer.sh 10 cpu
